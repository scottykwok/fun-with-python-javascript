import asyncio from js import document from pyodide.http import open_url from pyodide.ffi import create_proxy def load_svg(elementId, url): container = document.getElementById(elementId) container.innerHTML = open_url(url).read() return container # Load SVG container = load_svg("container", "images/python-hk.svg") # On Click callback def on_click(event): print(f"{event.currentTarget.id} is clicked!") container.querySelector('#'+event.currentTarget.id).classList.toggle('flip') # Register buttons for c in "pycon-hk": button = container.querySelector(f"#mj-{c}") button.classList.add("clickable") button.addEventListener("click", create_proxy(on_click))
