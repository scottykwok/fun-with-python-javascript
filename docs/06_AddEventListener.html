Python is clicked!") for i in "pycon-hk": container.querySelector(f"#mj-{i}").classList.toggle('flip') else: container.querySelector('#'+c).classList.toggle('flip') # Register buttons for i in "pycon-hk": button = container.querySelector(f"#mj-{i}") button.classList.add("clickable") button.addEventListener("click", create_proxy(on_click))
